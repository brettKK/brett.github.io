import "fmt"
type Heap struct {
    Data []int
    N int
}
func NewHeap() heap {
	return heap{
		Data: make([]int, 0),
		N: 0,
	}
}
// 入堆
func (h *heap) Add(elem int) {
	h.N++
	if len(h.Data) == 0 {
		h.Data =  append(h.Data, 0, elem)
		return
	} else {
		h.Data = append(h.Data, elem)
		// Upper Change heap
		h.UpperChange(len(h.Data)-1)
	}
}
func (h *heap) UpperChange(index int) {
	for index > 1 {
		parent := h.getParent(index)
		if h.Data[parent] > h.Data[index] {
			break
		} else {
			h.swap(parent, index)
			index = parent
		}

	}
}
// 出堆
func (h *heap) DelMax() int {
	if len(h.Data) < 1 {
		return -1
	} else {
		max := h.Data[1]
		h.Data[1] = h.Data[len(h.Data) -1]
		h.N--
		h.Data = h.Data[:len(h.Data) - 1]
		//down change
		h.DownChange(1)
		return max
	}
	return -1
}
func (h *heap) DownChange(index int) {
	for h.getLeftChld(index) <= h.N &&h.getLeftChld(index) >= 1 && h.getRightChld(index) <= h.N && h.getRightChld(index) >= 1{
		cur := h.getLeftChld(index)
		if h.Data[h.getRightChld(index)] > h.Data[cur] {
			cur = h.getRightChld(index)
		}
		h.swap(index, cur)
		index = cur
	}
}

func (h *heap) getParent(chld int) int {
	return chld /2
}
func (h *heap) getLeftChld(parent int) int {
	if h.N < parent * 2 {
		return -1
	}
	return parent * 2
}
func (h *heap) getRightChld(parent int) int {
	if h.N < parent * 2 + 1 {
		return -1
	}
	return parent * 2 + 1
}

func quick_sort_simple(arr []int) {
    if len(arr) < 2 {
        return
    }
	// index 是小于start的最后一个元素地址，也就是partition
    var start, index int = 0, 0
    for i := 1; i < len(arr); i++ {
        if arr[i] < arr[start] {
            index++
            arr[i], arr[index] = arr[index], arr[i]
        }
    }
    arr[start], arr[index] = arr[index], arr[start]
    quick_sort_simple(arr[start:index])
    quick_sort_simple(arr[index+1:])
}

归并排序 🔗

func merge_sort(arr []int) []int{
	if len(arr) < 2 {return arr}
	mid := len(arr)>>1
	left_arr := merge_sort(arr[:mid])
	right_arr := merge_sort(arr[mid:])
	return merge(left_arr, right_arr)
}

func merge(left, right []int) []int {
	var res []int = make([]int, len(left) + len(right))
	var k, i, j int
	for i <len(left) && j < len(right) {
		if left[i] <= right[j] {
			res[k] = left[i]
			i++ 
		} else {
			res[k] = right[j]
			j++
		}
		k++
	}
	for i < len(left) {
		res[k] = left[i]
		k++; i++
	}
	for j < len(right) {
		res[k] =right[j]
		k++; j++
	}
	return res
}

type LinkNode struct {
    key, val int 
    pre, next *LinkNode
}

type LRUCache struct {
    m map[int]*LinkNode // 指向哈希表的指针
    cap int // 长度
    head, tail *LinkNode // 两个哨兵
}

func Constructor(capacity int) LRUCache {
    head := &LinkNode{0, 0, nil, nil}
    tail := &LinkNode{0, 0, nil, nil}
    head.next = tail
    tail.pre = head
    return LRUCache{make(map[int]*LinkNode), capacity, head, tail}
}


func (this *LRUCache) Get(key int) int {
    cache := this.m
    if node, exist := cache[key]; exist {
        this.moveToHead(node)
        return node.val
    }else {
        return -1
    }
}
func (this *LRUCache) moveToHead(node *LinkNode) {
    head := this.head
    // 从当前位置删除节点
    node.pre.next = node.next
    node.next.pre = node.pre
    // 将节点插入头部
    node.next = head.next
    head.next.pre = node
    node.pre = head
    head.next = node
}

func (this *LRUCache) Put(key int, value int)  {
    head := this.head
    tail := this.tail
    cache := this.m
    if node, exist := cache[key]; exist {// 如果已经存在对应的节点，要将对应的元素交换到头部；并更新值（值可能变了）
        node.val = value
        this.moveToHead(node)
    }else {// 节点不存在
        // 创建节点
        node := &LinkNode{key, value, nil, nil}
        // 判断缓存容量是否已经满了
        if len(cache) == this.cap {
            // 删除最后的元素
            delete(cache, tail.pre.key)
            tail.pre.pre.next = tail
            tail.pre = tail.pre.pre
        }
        // 将节点放到头部
        node.next = head.next
        node.pre = head
        head.next.pre = node
        head.next = node
        // 参入哈希表中
        cache[key] = node
    }
}

func (sl *zskiplist) Search(score int) bool {
    cur := sl.head
    for n := sl.level - 1; n >= 0; n-- {
        for cur.level[n] != nil {
            if cur.level[n].score == score {
                return true
            } else if cur.level[n].score > score {
                // 向下寻找
                break
            } else {
                // 向前找

    }
    return false

func NewSkipList() *SkipList {
    return &SkipList {

        height: 1;
        length: 0;
    }

func NewNode(levels, value uint32) *Node {
    node := new(Node)
    node.value = value
    node.levels = make([]*Level, levels)
    for i := 0; i < levels; i++ {
        node.levels[i] = new(Level)
    }
    return node
}
func (sl *SkipList) randomLevel() int {
    level := 1
    r := rand.New()
    for r.Float64() < p && level < MaxLevel {
        level++
    }
    return level
}

        update[i] = tmp
        // 遍历完一层后，下一层的起始查找位置一定在上一层tmp之后        
    }
    level := sl.randomLevel()
    node := NewNode(uint32(level), value)1
    if uint32(level) > sl.height {
        sl.height = uint32(level)
    }
    // 从 0- level层的链表均需要更新， level+1 ～ height之间的上层链表不用更新
    for i := 0; i < level; i++ {
        // 新节点超过height，将头节点对应层数的后继节点设置为新节点 
        if update[i] == nil {
            sl.header.levels[i].next = node
            continue

        // 修改新节点的后继节点为

        // 修改每一层最后一个比value 小的节点的后继节点为新节点
        update[i].levels[i].next = node
    }
    sl.length++
    return true
}
