type zskiplist struct {
    head *zskiplistNode
    tail *zskiplistNode
    length int
    level int
}
type zskiplistNode struct {
    level []*zskiplistLevel
    backward *zskiplistNode
    score int
    obj interface{}
}
type zskiplistLevel struct{
    forward *zskiplistNode
    span int
}

// search key
func (sl *zskiplist) Search(score int) bool {
    cur := sl.head
    for n := sl.level - 1; n >= 0; n-- {
        for cur.level[n] != nil {
            if cur.level[n].score == score {
                return true
            } else if cur.level[n].score > score {
                // 向下寻找
                break
            } else {
                // 向前找
                cur = cur.level[n]
            }
        }
    }
    return false
}

// add key
func (sl *zskiplist) Add(score int, value interface{}) {
    // update_array
    update := make([]*zskiplistNode, sl.level)

    // update
    level := randLevel()
    if level > sl.level {
        for i := sl.level; i < level; i++ {
            update[i] = sl.head
        }
        sl.level = level
    }
    node := newNode(score, value, level)
    for i := 0; i < level; i++ {
        node.level[i] = update[i].level[i]
        update[i].level[i] = node
    }
}
