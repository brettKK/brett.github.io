why PL need closure? 🔗

想一下在函数式编程语言中，宣传函数是一等公民，把函数作为一种类型定义变量，把函数作为函数的参数和返回值时。那么我们定义一个函数时，很自然出来一个问题：函数可以使用哪些变量？

  • 函数自己的实参
  • 函数内定义的局部变量
  • 程序整个生命周期内都一直存活的全局静态变量
  • Outer函数A定义Inner函数B，并把Inner函数B作为Outer函数A的返回值， Inner函数B中能使用scope A中的变量吗？
    • 当然可以，站在Inner函数B的角度下，Inner函数B出生在Outer函数A里面（出生==首次被定义），自然看到函数A的内部变量了。
    • Outer函数A对应的scope(作用域)下有的东西可以被Inner函数B可见可使用。
    • 但是Inner函数B作为了返回值，会在未来的某个程序点处使用。
    • 此时Outer函数A的scope中实参和局部变量在栈上会消失，所以出现"闭包"概念就构造一个结构体去capture 外部变量。

所以，闭包是为了解决函数式语言中函数在使用变量时能看到哪些变量（变量作用域）的技术。

正则表达式引擎 🔗

regular expression engine 有限自动机Finite-state machine

Kleene’s Theorem：正则表达式与有限自动机（NFA和DFA）等价且可以相互转换。

Ken_Thompson在1968年文章《Regular Expression Search Algorithm》 实现了从regular expression转换到FA的算法。

Union, Concatenation, Closure (Kleene Star)， todo画状态机描述下如何通过叠加最小的单元状态机去构造复杂的状态机。

Thompson’s Construction思想是：

  • 将infix的正则表达式变为postfix表达式(逆波兰式很方便计算机的栈式计算)，方便后续不用在考虑优先级的问题。
  • 将正则表达式分解为最小的子表达式，再通过最小的有限自动机单元（识别零个字符 $\epsilon$ 和 识别一个字符的最小单元）构造完整的FA。
type State struct {
    isEnd bool,                   // 状态是否为终结可接受的状态
    transition map[byte]*State,   // 遇到symbol后next state的存储
    epsilonTransitions: []*State, // 不消耗symbol可以前往哪些next state
}
func addEpsilonTransition(from, to *State) {
    append(from.epsilonTransitions, to)
}
func addTransition(from, to *State, symbol byte) {
    from.transition[symbol] = to
}

// NFA 是一个只包含状态起始状态指针和终止状态指针的结构体。
type NFA struct {
    start *State, end *State
}

// 构造最基本的NFA an ε-NFA， a symbol-NFA. 
func fromEpsilon() *NFA{
    from := &State{ isEnd: false, transition: make(map[byte]*State), epsilonTransitions: make([]*State) }
    to := &State { isEnd: true, transition: make(map[byte]*State), epsilonTransitions: make([]*State) }
    addEpsilonTransition(from, to)
    return &NFA{start: from, end: to}
}
fun fromSymbol(symbol byte) *NFA{
    from := &State{ isEnd: false, transition: make(map[byte]*State), epsilonTransitions: make([]*State) }
    to := &State { isEnd: true, transition: make(map[byte]*State), epsilonTransitions: make([]*State) }}
    addTransition(from, to, symbol)
    return &NFA{start: from, end: to}
}

// 接着可以构造稍微复杂点的NFA， 或， 与， star *
func union(first, second *NFA) *NFA {
    start := &State{isEnd: false, transition: make(map[byte]*State), epsilonTransitions: make([]*State)}
    addEpsilonTransition(start, first.start)
    addEpsilonTransition(start, second.start)
    end := &State{isEnd: true, transition: make(map[byte]*State), epsilonTransitions: make([]*State) }
    addEpsilonTransition(first.end, end)
    first.end.isEnd = false
    addEpsilonTransition(second.end, end)   
    second.end.isEnd = false
    return &NFA{start, end}
}
func concat(first, second *NFA) *NFA {
    addEpsilonTransition(first.end, second.start)
    first.end.isEnd = false
    return &NFA{start: first.start, end: second.end}
}

// infix RE: (a|b)*c -> postfix RE
// 显示插入连接符.      (a|b)*c 改为了 （a|b)*.c
// infix RE 改为 postfix。     (a|b)*.c 改为了  ab|*c.
// 构造NFA
func toNFA(postfixExp []byte) *NFA {
    if len(postfixExp) == 0 { return fromEpsilon()}
    stack := make([]*NFA, 0)
    for _, token := range postfixExp {
        switch token {
        case '*':
            nfa := stack[len(stack) - 1]   // peek the top on the stack
            stack = stack[ :len(stack) - 1] // pop the stack
            stack = append(stack, closure(nfa))
        case '|':
            rightNFA := stack[len(stack) - 1]
            stack = stack[ :len(stack) - 1]
            leftNFA := stack[len(stack) - 1]
            stack = stack[:len(stack) - 1]
            stack = append(stack, union(leftNFA, rightNFA)) //
        case '.':
            rightNFA := stack[len(stack) - 1]
            stack = stack[ :len(stack) - 1]
            leftNFA := stack[len(stack) - 1]
            stack = stack[:len(stack) - 1]
            stack = append(stack, concat(leftNFA, rightNFA))
        default:
            stack = append(stack, fromSymbol(token))
        }
    }
    return stack[len(stack) - 1] // 栈顶为最终的NFA
}
